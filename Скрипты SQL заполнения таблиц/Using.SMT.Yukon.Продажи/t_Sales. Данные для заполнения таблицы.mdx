SELECT
	{
		[Measures].[Количество]
		,[Measures].[Сумма продаж]
	} ON COLUMNS
	,NON EMPTY (
		FILTER(([13 Документы продаж].[ID].[ID]
		,[13 Документы продаж].[Номер документа].[Номер документа]
		,[13 Документы продаж].[Дата документа].[Дата документа]
//		,[05 Контрагенты].[ID].[ID]
//		,[06 Торговые точки].[ID].[ID]
//		,[07 Маршруты].[ID].[ID]
		,[13 Документы продаж].[Дата доставки].[Дата доставки]
		,[13 Документы продаж].[Дата оплаты].[Дата оплаты]) , NOT ISEMPTY([13 Документы продаж].[ID].[ID]))
	) ON ROWS
FROM [Sales]
WHERE ([07-01-01 Системы мобильной торговли].[Система мобильной торговли].&[Yukon-PG])
